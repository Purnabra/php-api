#git start
git init
#git commit (means something changes or add something)
git commit -m "your_message"
#add file in git
git add .(for all files) or git add yout_file_name.txt
#go to the desired branch
git checkout your_branch_name
#to create a new brach locally
git checkout -b your_branch_name
#push it to the github repo
git push -u origin your_branch_name --force(for forcefully push)
#how to pull a file and download it in local system
git checkout origin/your_branch_name -- any_file_name.extension
